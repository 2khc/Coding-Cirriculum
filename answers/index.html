Arrays

  #include <stdio.h>

  int main (void) {
      float numbers[15];
      printf("Enter 15 floating-point values.");

      for(int i=0; i<15; ++i)
      scanf("%f", &numbers[i]); //store all the entered floating-point values into an array

      float sum;

      for(int j=0; j<15; ++j)
      sum += numbers[j];

      printf("The sum of an array of 15 floating-point values is %f", sum);

      return 0;
  }

Making Decisions 1

 #include <stdio.h>

 int main (void) {
     float value1, value2;
     char operator;

     printf("Type in your expression.\n");
     scanf("%f %c %f", &value1, &operator, &value2);

     if (operator == '+')
         printf("%.2f\n", value1 + value2);

     else if (operator == '-')
         printf("%.2f\n", value1 - value2);

     else if (operator == '*')
         printf("%.2f\n", value1 * value2);

     else if (operator == '/')
         if (value2 == 0)
             printf("Division by zero.\n");
         else
             printf("%.2f\n", value1 / value2);

     else
         printf("Unknown operator.\n"); //in case an unknown operator is typed in.

     return 0;

 }

Making Decisions 2

 #include <stdio.h>

 int main (void) {
     float value1, value2;
     char operator;

     printf("Type in your expression.\n");
     scanf("%f %c %f", &value1, &operator, &value2);

     switch (operator) {
         case '+':
             printf("%.2f\n", value1 + value2);
             break;

         case '-':
             printf("%.2f\n", value1 - value2);
             break;

         case '*':
             printf("%.2f\n", value1 * value2);
             break;

         case '/':
             if (value2 == 0)
                 printf("Division by zero.\n");
             else
                 printf("%.2f\n", value1 / value2);
             break;

         default:
             printf("Unknown operator.\n");
             break;
     }

     return 0;

 }
