Chapter 3: Arrays

  #include <stdio.h>

  int main (void) {
      float numbers[15];
      printf("Enter 15 floating-point values.");

      for(int i=0; i<15; ++i)
      scanf("%f", &numbers[i]); //store all the entered floating-point values into an array

      float sum;

      for(int j=0; j<15; ++j)
      sum += numbers[j];

      printf("The sum of an array of 15 floating-point values is %f", sum);

      return 0;
  }

Making Decisions

  #include <stdio.h>

  int main (void)
  {
      float value1, value2;
      char operation;

      printf("Type in your expression.\n");
      scanf("%f %c %f", &value1, &operation, &value2);

      if (operation == '+')
      {
          printf("%.2f\n", value1 + value2);  
      }
      else if (operation == '-')
      {
          printf("%.2f\n", value1 - value2);  
      }
      else if (operation == '*')
      {
           printf("%.2f\n", value1 * value2);
      }
      else if (operation == '/')
      {
          if (value2 == 0)
          {
            printf("Division by zero.\n");            
          }
          else
          {
          printf("%.2f\n", value1 / value2);
          }          
      }         
      else
      {
          printf("Unknown operation.\n"); //in case an unknown operation is typed in.");
      }

      return 0;
  }
